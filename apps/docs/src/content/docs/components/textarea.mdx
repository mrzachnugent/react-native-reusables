## Usage

```tsx
import { Textarea } from '~/components/ui/textarea';
import { Text } from '~/components/ui/text';

function Example() {
  const [value, setValue] = React.useState('');
  return (
      <Textarea
        ref={inputRef}
        placeholder='Write some stuff...'
        value={value}
        onChangeText={setValue}
        aria-labelledby='textareaLabel'
      />
  );
}
```
