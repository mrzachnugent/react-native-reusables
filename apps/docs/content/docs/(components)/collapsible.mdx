### Usage

```tsx
import { Collapsible } from "@/components/ui/collapsible"
import { Text } from "@/components/ui/text"

```

```tsx
<Collapsible>
  <CollapsibleTrigger>
    <Text>Can I use this in my project?</Text>
  </CollapsibleTrigger>
  <CollapsibleContent>
    <Text>Yes. Free to use for personal and commercial projects. No attribution
    required.</Text>
  </CollapsibleContent>
</Collapsible>
```
